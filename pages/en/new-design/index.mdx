    ```js displayName="Create an HTTP Server"
    import { createServer } from 'node:http';

    const server = createServer((req, res) => {
      res.writeHead(200, { 'Content-Type': 'text/plain' });
      res.end('Hello World!');
    });

    server.listen(3000, '127.0.0.1', () => {
      console.log('Listening on 127.0.0.1:3000');
    });
    ```

    ```js displayName="Read and Write Files"
    import { readFile, writeFile } from 'node:fs/promises';

    const content = {
      hello: { world: true },
    };

    await writeFile('file.txt', JSON.stringify(content));

    // yields {"hello":{"world":true}}
    const file = await readFile('file.txt', 'utf8');
    ```

    ```js displayName="Write Tests"
    import assert from 'node:assert';
    import test from 'node:test';

    test('that 1 is equal 1', () =>
      assert.strictEqual(1, 1));

    test('that throws as 1 is not equal 2', () => {
      // throws an exception because 1!=2
      assert.strictEqual(1, 2);
    });
    ```

    ```js displayName="Hash a File"
    import { createHash } from 'node:crypto';
    import { readFile } from 'node:fs/promises';

    const hasher = createHash('sha1');

    hasher.setEncoding('hex');
    hasher.write(await readFile('package.json'));
    hasher.end();

    const fileHash = hasher.read();
    ```

    ```js displayName="Read a Stream"
    import { createReadStream, createWriteStream } from 'node:fs';

    const res = await fetch('https://nodejs.org/dist/index.json');
    const json = await res.json(); // Parsed JavaScript object (JSON)

    readableStream.setEncoding('utf8');
    readableStream.on('data', chunk => writableStream.write(chunk));
    ```
